    <script>
        // show the selected tab
        function showTab(tab) {
            // hide all forms
            document.querySelectorAll('.form-section').forEach(function(el) {
                el.classList.remove('active');
            });
            // remove active from all tab buttons
            document.querySelectorAll('.tab-btn').forEach(function(el) {
                el.classList.remove('active');
            });

            // show selected form
            document.getElementById(tab).classList.add('active');

            // highlight correct tab button
            if (tab === 'signin') {
                document.querySelectorAll('.tab-btn')[0].classList.add('active');
            } else {
                document.querySelectorAll('.tab-btn')[1].classList.add('active');
            }
        }

        // handle sign in
        function handleSignIn(e) {
            e.preventDefault();

            var username = document.getElementById('signin-username').value;
            var password = document.getElementById('signin-password').value;
            var hasError = false;

            document.getElementById('signin-user-err').style.display = 'none';
            document.getElementById('signin-pass-err').style.display = 'none';
            document.getElementById('signin-msg').style.display = 'none';

            if (username == '') {
                document.getElementById('signin-user-err').style.display = 'block';
                hasError = true;
            }

            if (password == '') {
                document.getElementById('signin-pass-err').style.display = 'block';
                hasError = true;
            }

            if (!hasError) {
                document.getElementById('signin-msg').style.display = 'block';
            }
        }

        // handle sign up
        function handleSignUp(e) {
            e.preventDefault();

            var username = document.getElementById('signup-username').value;
            var password = document.getElementById('signup-password').value;
            var hasError = false;

            document.getElementById('signup-user-err').style.display = 'none';
            document.getElementById('signup-pass-err').style.display = 'none';
            document.getElementById('signup-msg').style.display = 'none';

            if (username == '') {
                document.getElementById('signup-user-err').style.display = 'block';
                hasError = true;
            }

            if (password == '') {
                document.getElementById('signup-pass-err').style.display = 'block';
                hasError = true;
            }

            if (!hasError) {
                document.getElementById('signup-msg').style.display = 'block';
            }
        }
    </script>
